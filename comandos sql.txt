CODIGOS SQLS PARA CRIAR AS TABELAS NO BANCO DE DADOS

* TABELA CLIENTE
CREATE TABLE CLIENTE(ID INT PRIMARY KEY AUTO_INCREMENT, NOME VARCHAR(255),CPF VARCHAR(255),TELEFONE VARCHAR(255),EMAIL VARCHAR(255),RUA VARCHAR(255),NUMERO VARCHAR(255),CIDADE VARCHAR(255),ESTADO VARCHAR(255));

* TABELA PEDIDO
CREATE TABLE PEDIDO(ID INT PRIMARY KEY AUTO_INCREMENT, CLIENTE VARCHAR(255), PRODUTO VARCHAR(255), METODOPAGAMENTO VARCHAR(255), STATUS VARCHAR(255));

*TABELA PRODUTO
CREATE TABLE PRODUTO(ID INT PRIMARY KEY AUTO_INCREMENT, NOMEPRODUTO VARCHAR(255), DESCRICAO VARCHAR(255), PRECO VARCHAR(255));

*TABELA FALE CONOSCO
CREATE TABLE FALECONOSCO(ID INT PRIMARY KEY AUTO_INCREMENT, NOME VARCHAR(255), EMAIL VARCHAR(255), DUVIDA VARCHAR(1000));

*TABELA LOGIN
CREATE TABLE LOGIN (CPF VARCHAR(255), SENHA VARCHAR(25));

*TABELA METODO PAGAMENTO 
CREATE TABLE METODOPAGAMENTO (IDMETODOPAGAMENTO INT, TITULAR VARCHAR(255),NUMCARTAO VARCHAR (16),VALIDADE VARCHAR (5), CODSEG VARCHAR(3));

*TABELA CARRINHO 
CREATE TABLE CARRINHO (ID INT PRIMARY KEY AUTO_INCREMENT, IDCLIENTE INT, IDPRODUTO INT)
