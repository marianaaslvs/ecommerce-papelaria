* TABELA PEDIDO
CREATE TABLE PEDIDO(ID INT PRIMARY KEY AUTO_INCREMENT, CLIENTE VARCHAR(255), PRODUTO VARCHAR(255), DATA DATE DEFAULT CURRENT_DATE, ENDERECO VARCHAR(255), VALOR DOUBLE, STATUS VARCHAR(255));

*TABELA PRODUTO
CREATE TABLE PRODUTO(ID INT PRIMARY KEY AUTO_INCREMENT, NOMEPRODUTO VARCHAR(255), DESCRICAO VARCHAR(255), PRECO DOUBLE);
ALTER TABLE PRODUTO ADD IMAGE VARCHAR (255);

*TABELA FALE CONOSCO
CREATE TABLE FALECONOSCO(ID INT PRIMARY KEY AUTO_INCREMENT, NOME VARCHAR(255), EMAIL VARCHAR(255), DUVIDA VARCHAR(1000));
ALTER TABLE FALECONOSCO ADD STATUS VARCHAR (255);


*TABELA METODO PAGAMENTO 
CREATE TABLE METODOPAGAMENTO (IDMETODOPAGAMENTO INT PRIMARY KEY AUTO_INCREMENT, TITULAR VARCHAR(255),NUMCARTAO VARCHAR (16),VALIDADE VARCHAR (5), CODSEG VARCHAR(3));

*TABELA CARRINHO 
CREATE TABLE CARRINHO (IDCARRINHO INT PRIMARY KEY AUTO_INCREMENT, IDCLIENTE VARCHAR(255), IDPRODUTO VARCHAR(255), QUANTIDADE INT)
